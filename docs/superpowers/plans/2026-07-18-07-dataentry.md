### Task 1: Models

```python
"""Data entry models."""
from django.db import models


class DataEntryRecord(models.Model):
    """Staged inbound record awaiting validation/dispatch.

    Attributes:
        tenant: Owning tenant.
        source: form | csv | email | webhook.
        status: pending | valid | invalid | dispatched | dead.
        raw: Original payload.
        mapped: Normalized fields after mapping.
        target_type: contact | invoice | other.
        error: Last validation error.
    """

    class Source(models.TextChoices):
        """Ingest sources."""

        FORM = "form", "Form"
        CSV = "csv", "CSV/Excel"
        EMAIL = "email", "Email"
        WEBHOOK = "webhook", "Webhook"

    class Status(models.TextChoices):
        """Record lifecycle."""

        PENDING = "pending", "Pending"
        VALID = "valid", "Valid"
        INVALID = "invalid", "Invalid"
        DISPATCHED = "dispatched", "Dispatched"
        DEAD = "dead", "Dead"

    tenant = models.ForeignKey("tenants.Tenant", on_delete=models.CASCADE)
    source = models.CharField(max_length=20, choices=Source.choices)
    status = models.CharField(
        max_length=12, choices=Status.choices, default=Status.PENDING
    )
    raw = models.JSONField(default=dict)
    mapped = models.JSONField(default=dict)
    target_type = models.CharField(max_length=40, default="contact")
    error = models.TextField(blank=True, default="")
    created_at = models.DateTimeField(auto_now_add=True)
```

### Task 3: Validate → map → dispatch pipeline

```python
"""Celery pipeline for staged records."""
from celery import shared_task

from dataentry.models import DataEntryRecord


@shared_task
def process_record(record_id: int) -> None:
    """Validate, map, and dispatch a staged record.

    Args:
        record_id: DataEntryRecord PK.
    """
    rec = DataEntryRecord.objects.get(pk=record_id)
    try:
        mapped = _map(rec)
        _validate(mapped, rec.target_type)
        rec.mapped = mapped
        rec.status = DataEntryRecord.Status.VALID
        rec.save(update_fields=["mapped", "status"])
        _dispatch(rec)
        rec.status = DataEntryRecord.Status.DISPATCHED
        rec.save(update_fields=["status"])
    except Exception as exc:  # noqa: BLE001
        rec.status = DataEntryRecord.Status.INVALID
        rec.error = str(exc)
        rec.save(update_fields=["status", "error"])
```

- [ ] `_map` applies tenant field map
- [ ] `_validate` checks required fields per target_type
- [ ] `_dispatch` creates Contact (onboarding.Client) or Invoice draft
- [ ] Tests + commit: `feat(dataentry): validate/map/dispatch pipeline`
