"""Follow-up models."""
from django.db import models


class FollowupRule(models.Model):
    """Reusable rule describing when/how to remind.

    Attributes:
        tenant: Owning tenant.
        name: Rule name.
        channel: email | internal_task | both.
        offset_days: Days relative to due date (negative = before).
        template: Email body template with {placeholders}.
    """

    class Channel(models.TextChoices):
        """Delivery channels."""

        EMAIL = "email", "Email"
        INTERNAL = "internal_task", "Internal task"
        BOTH = "both", "Both"

    tenant = models.ForeignKey("tenants.Tenant", on_delete=models.CASCADE)
    name = models.CharField(max_length=100)
    channel = models.CharField(max_length=20, choices=Channel.choices)
    offset_days = models.IntegerField(default=-7)
    template = models.TextField(default="Reminder: {subject} is due on {due_date}.")


class Reminder(models.Model):
    """A concrete reminder instance.

    Attributes:
        tenant: Owning tenant.
        rule: Optional rule that produced this.
        subject: Short subject.
        due_at: When to fire.
        recipient_email: Email target (if channel includes email).
        status: pending | sent | cancelled.
        context: JSON for template fill.
    """

    class Status(models.TextChoices):
        """Reminder lifecycle."""

        PENDING = "pending", "Pending"
        SENT = "sent", "Sent"
        CANCELLED = "cancelled", "Cancelled"

    tenant = models.ForeignKey("tenants.Tenant", on_delete=models.CASCADE)
    rule = models.ForeignKey(
        FollowupRule, null=True, blank=True, on_delete=models.SET_NULL
    )
    subject = models.CharField(max_length=200)
    due_at = models.DateTimeField(db_index=True)
    recipient_email = models.EmailField(blank=True, default="")
    status = models.CharField(
        max_length=12, choices=Status.choices, default=Status.PENDING
    )
    context = models.JSONField(default=dict)
    created_at = models.DateTimeField(auto_now_add=True)
```

### Task 2: Services

"""Follow-up services."""
from datetime import datetime, timedelta

from django.utils import timezone

from followups.models import FollowupRule, Reminder
from invoices.models import Invoice


def schedule_invoice_reminder(invoice: Invoice) -> Reminder:
    """Schedule a reminder 7 days before the invoice due date.

    Args:
        invoice: Issued invoice.

    Returns:
        Created Reminder.
    """
    due_at = timezone.make_aware(
        datetime.combine(invoice.due_date - timedelta(days=7), datetime.min.time())
    )
    return Reminder.objects.create(
        tenant=invoice.tenant,
        subject=f"Invoice {invoice.number} due",
        due_at=due_at,
        recipient_email=invoice.client.email,
        context={
            "subject": f"Invoice {invoice.number}",
            "due_date": invoice.due_date.isoformat(),
            "number": invoice.number,
        },
    )


def process_due_reminders() -> int:
    """Send all pending reminders whose due_at has passed.

    Returns:
        Number of reminders sent.
    """
    from integrations.models import IntegrationConfig
    from integrations.services import get_adapter
    from audit.services import log as audit_log

    now = timezone.now()
    qs = Reminder.objects.filter(status=Reminder.Status.PENDING, due_at__lte=now)
    count = 0
    for rem in qs.select_related("tenant"):
        body = (
            rem.rule.template.format(**rem.context)
            if rem.rule
            else f"Reminder: {rem.subject}"
        )
        if rem.recipient_email:
            email = get_adapter(rem.tenant, IntegrationConfig.Kind.EMAIL)
            email.send(
                to=[rem.recipient_email],
                subject=rem.subject,
                body=body,
            )
        rem.status = Reminder.Status.SENT
        rem.save(update_fields=["status"])
        audit_log(
            tenant=rem.tenant,
            actor=None,
            event="followup.sent",
            payload={"reminder_id": rem.id},
        )
        count += 1
    return count
```
